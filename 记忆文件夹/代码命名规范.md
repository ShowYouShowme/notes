```ini
[文件名]
desc = 全小写,不同单词使用_分隔

[目录名]
desc = 全小写,不同单词使用_分隔

[mysql]
desc = 库名，表名，字段名必须使用小写，用下划线分隔

[protobuf]
desc = 和 Class 一样

[class]
类名           =  大驼峰
public  属性名  = 小驼峰
private 属性名  = 小驼峰
public  方法名  = 小驼峰
private 方法名  = 小驼峰

[命名空间]
desc = 大驼峰

[函数名]
desc = 小驼峰

[变量名]
; 疑问
desc = 小驼峰

[常量名]
desc = 大写字母和下划线; MAX_VALUE

[接口]
desc = I开头,然后是和class一样
```
